using MySql.Data.MySqlClient;
using System.Data;
using System.Collections.Generic;
using System;

namespace Media_diary
{
    public class DBClass
    {
        private readonly string connectionString = "server=localhost;user=root;database=media_reviews;port=3306;password=root";
        public int Execute(string query, Dictionary<string, object> parameters = null)
        {
            using (MySqlConnection connection = new MySqlConnection(connectionString))
            {
                connection.Open();
                using (MySqlCommand command = new MySqlCommand(query, connection))
                {
                    if (parameters != null)
                    {
                        foreach (var param in parameters)
                        {
                            command.Parameters.AddWithValue(param.Key, param.Value);
                        }
                    }
                    return command.ExecuteNonQuery();
                }
            }
        }
        public DataTable Select(string query, Dictionary<string, object> parameters = null)
        {
            DataTable table = new DataTable();
            using (MySqlConnection connection = new MySqlConnection(connectionString))
            {
                connection.Open();
                using (MySqlCommand command = new MySqlCommand(query, connection))
                {
                    if (parameters != null)
                    {
                        foreach (var param in parameters)
                        {
                            command.Parameters.AddWithValue(param.Key, param.Value);
                        }
                    }
                    using (MySqlDataAdapter adapter = new MySqlDataAdapter(command))
                    {
                        adapter.Fill(table);
                    }
                }
            }
            return table;
        }
    }
}

namespace Media_diary

    public partial class CaptchaForm : Form
    {
        private PictureBox[] slots;
        private PictureBox firstSelected;
        private int fails = 0;
        int currentUserId;
        public CaptchaForm(int userId)
        {
            InitializeComponent();
            this.Load += new System.EventHandler(this.captchaForm_Load);

            slots = new[] { pb1, pb2, pb3, pb4 };

            foreach (var pb in slots)
                pb.Click += Slot_Click;
            currentUserId = userId;
        }

        private void captchaForm_Load(object sender, EventArgs e)
        {
            for (int i = 0; i < slots.Length; i++)
                slots[i].Tag = i;

            ShuffleImages();
        }

        private void ShuffleImages()
        {
            var parts = new (Image img, int id)[slots.Length];

            for (int i = 0; i < slots.Length; i++)
                parts[i] = ((Image)slots[i].Image.Clone(), (int)slots[i].Tag);

            Random r = new Random();

            for (int i = 0; i < parts.Length; i++)
            {
                int j = r.Next(parts.Length);
                (parts[i], parts[j]) = (parts[j], parts[i]);
            }

            for (int i = 0; i < slots.Length; i++)
            {
                slots[i].Image = parts[i].img;
                slots[i].Tag = parts[i].id;
            }
        }

        private void Slot_Click(object sender, EventArgs e)
        {
            var clicked = (PictureBox)sender;

            if (firstSelected == null)
            {
                firstSelected = clicked;
                clicked.BorderStyle = BorderStyle.Fixed3D;
                return;
            }

            if (firstSelected != clicked)
            {
                (firstSelected.Image, clicked.Image) = (clicked.Image, firstSelected.Image);
                (firstSelected.Tag, clicked.Tag) = (clicked.Tag, firstSelected.Tag);
            }

            firstSelected.BorderStyle = BorderStyle.None;
            firstSelected = null;
        }

        private void btnCaptchaCheck_Click(object sender, EventArgs e)
        {
            bool ok = true;

            for (int i = 0; i < slots.Length; i++)
                if ((int)slots[i].Tag != i)
                    ok = false;

            if (ok)
            {
                MessageBox.Show("Капча пройдена!");
                DialogResult = DialogResult.OK;
                Close();
                return;
            }

            fails++;

            if (fails >= 3)
            {
                DBClass db = new DBClass();
                string query = "UPDATE user SET is_blocked = 1 WHERE user_id = @currentUserId";
                var parameters = new Dictionary<string, object>
                {
                    { "@currentUserId", currentUserId }
                };
                db.Execute(query, parameters);
                MessageBox.Show("Вы заблокированы. Обратитесь к администратору");
                DialogResult = DialogResult.No;
                Close();
            }
            else
            {
                MessageBox.Show($"Неверно. Попыток осталось: {3 - fails}");
            }
        }
    }
}

        private void LoadMediaFromCollection()
        {

            {
                { "@collectionId", collectionId }
            };
            DataTable mediaFromCollection = db.Select(MFSQuery, MFSParameters);

            DataTable collection = new DataTable();

            if (mediaFromCollection.Rows.Count > 0)
            {
                int firstMediaId = Convert.ToInt32(mediaFromCollection.Rows[0][0]);

                var firstMediaParameters = new Dictionary<string, object>
                {
                    { "@firstMediaId", firstMediaId }
                };
                DataTable firstMedia = db.Select(firstMediaQuery, firstMediaParameters);
                collection = firstMedia.Clone(); // клонируем структуру таблицы
            }
                        
            for (int i = 0; i < mediaFromCollection.Rows.Count; i++) // собираем все медиа в одну таблицу
            {
                int mediaId = Convert.ToInt32(mediaFromCollection.Rows[i][0]);

                var mediaParameters = new Dictionary<string, object>
                {
                    { "@mediaId", mediaId }
                };
                DataTable media = db.Select(mediaQuery, mediaParameters);

                foreach (DataRow row in media.Rows)
                {
                    collection.ImportRow(row);
                }
            }

            dgvCollection.DataSource = collection;
        }

        private void btnLogin_Click(object sender, EventArgs e)
        {
            string username = tbLogin.Text.Trim();
            string password = tbPassword.Text.Trim();

            if (string.IsNullOrEmpty(username) || string.IsNullOrEmpty(password))
            {
                MessageBox.Show("Введите логин и/или пароль.");
                return;
            }

            try // ищем пользователя
            {
                DBClass db = new DBClass();
                string query = "SELECT * FROM user WHERE username = @username";
                var parameters = new Dictionary<string, object>
                {
                    { "@username",  username }
                };
                DataTable user = db.Select(query, parameters);
                if (user.Rows.Count == 0)
                {
                    MessageBox.Show("Такого пользователя не существует.");
                    return;
                }
                string passwordHash = user.Rows[0]["password"].ToString();
                if (!BCrypt.Net.BCrypt.Verify(password, passwordHash))
                {
                    MessageBox.Show("Неверный логин или пароль. Проверьте корректность ввода или обратитесь к администратору.");
                    return;
                }
                int userId = Convert.ToInt32(user.Rows[0]["user_id"]);
                string userRole = user.Rows[0]["role"].ToString().Trim();
                int isBlocked = Convert.ToInt32(user.Rows[0]["is_blocked"]);
                if (isBlocked == 1)
                {
                    MessageBox.Show("Ваш аккаунт заблокирован. Обратитесь к администратору.");
                    return;
                }
                else
                {
                    if (userRole == "admin")
                    {
                        Hide();

                        MainAdminForm form = new MainAdminForm(userId);
                        form.ShowDialog();

                        Show();
                    }
                    else if (userRole == "user")
                    {
                        CaptchaForm captchaForm = new CaptchaForm(userId);
                        if (captchaForm.ShowDialog() == DialogResult.OK)
                        {
                            Hide();

                            MainUserForm form = new MainUserForm(userId);
                            form.ShowDialog();

                            Show();
                        }
                        else if (captchaForm.DialogResult == DialogResult.No)
                        {
                            MessageBox.Show("Ваш аккаунт заблокирован. Обратитесь к администратору.");
                        }
                        else { }
                    }
                }
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
        }

            DBClass db = new DBClass();
            string nameCollection = "SELECT name_collection AS 'Название подборки' FROM collection WHERE user_id_collection = @currentUserId OR user_id_collection = 1";
            var ncparameters = new Dictionary<string, object>
            {
                { "@currentUserId", currentUserId }

            DataTable table = db.Select(nameCollection, ncparameters);
            dgvCollection.DataSource = table;
            string recQuery = @"SELECT 
    m.media_id,
    m.title AS 'Название',
    m.genre AS 'Жанр',
    m.author AS 'Автор',
    COUNT(*) AS score
FROM
    media m
        JOIN
    review r ON r.user_id = @currentUserId
        JOIN
    media reviewed_media ON reviewed_media.media_id = r.media_id_review
WHERE
    (m.genre = reviewed_media.genre
        OR m.author = reviewed_media.author
        OR m.type = reviewed_media.type)
        AND m.media_id NOT IN (SELECT 
            media_id_review
        FROM
            review
        WHERE
            user_id = @currentUserId)
GROUP BY m.media_id , m.title , m.genre , m.author
ORDER BY score DESC
LIMIT 3;";
            var recParameters = new Dictionary<string, object>
            {
                { "@currentUserId", currentUserId }
            };
            DataTable rec = db.Select(recQuery, recParameters);
            if (rec.Rows.Count == 0)
            {
                label2.Text = "Здесь будут ваши рекомендации:";
                return;
            }
            dgvRecommend.DataSource = rec;
            dgvRecommend.Columns["media_id"].Visible = false;
            dgvRecommend.Columns["score"].Visible = false;

using System.Windows.Forms;

    public partial class RegisterForm : Form
    {
        public RegisterForm()
        {
            InitializeComponent();
        }

        private void btnRegister_Click(object sender, EventArgs e)
        {
            string login = tbLogin.Text.Trim();
            string username = tbUsername.Text.Trim();
            string password = tbPassword.Text.Trim();
            string repeatPassword = tbRepeatPassword.Text.Trim();

            if (string.IsNullOrWhiteSpace(login) || string.IsNullOrWhiteSpace(username) || string.IsNullOrWhiteSpace(password) || string.IsNullOrWhiteSpace(repeatPassword))
            {
                MessageBox.Show("Введите логин и/или пароль.");
                return;
            }
            if (password != repeatPassword)
            {
                MessageBox.Show("Пароли не совпадают.");
                return;
            }
            try
            {
                string passwordHash = BCrypt.Net.BCrypt.HashPassword(password);
                DBClass db = new DBClass();
                string checkQuery = "SELECT COUNT(*) FROM user WHERE username = @username;";
                var checkParameters = new Dictionary<string, object>
                {
                    { "@username", username }
                };
                DataTable table = db.Select(checkQuery, checkParameters);
                int userCount = Convert.ToInt32(table.Rows[0][0]);
                if (userCount > 0)
                {
                    MessageBox.Show("Пользователь с таким именем уже существует.");
                    return;
                }
                string query = "INSERT INTO user (login, username, password) VALUES (@login, @username, @passwordHash);";
                var parameters = new Dictionary<string, object>
                {
                    { "@login", login},
                    { "@username", username },
                    { "@passwordHash", passwordHash }
                };
                db.Execute(query, parameters);
                string selectQuery = "SELECT user_id FROM user WHERE username = @username AND password = @passwordHash;";
                var selectParameters = new Dictionary<string, object>
                {
                    { "@username", username },
                    { "@passwordHash", passwordHash }
                };
                int userId = db.Select(selectQuery, selectParameters).Rows[0].Field<int>("user_id");
                MessageBox.Show("Регистрация прошла успешно!");
                this.Hide();

                MainUserForm form = new MainUserForm(userId);
                form.ShowDialog();

                this.Show();
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка при регистрации: " + ex.Message);
            }
        }
